// Sistema migrazione completo class DataMigration { parseLegacyFormat(data) { // Formato: NAME|EMAIL|PHONE|DATE const pattern = /^ (?<name>[^|]+)\| (?<email>[\w.-]+@[\w.-]+)\| (?<phone>\d{10})\| (?<date>\d{2}\/\d{2}\/\d{4}) $/mx; const records = []; data.split('\n').forEach(line => { const match = line.match(pattern); if (match) records.push(match.groups); }); return records; } validateAndTransform(record) { // Valida email const emailValid = /^ [\w.-]+@ (?!tempmail|test) [\w.-]+\.[a-z]{2,} $/ix.test(record.email); // Trasforma data IT ‚Üí ISO const isoDate = record.date.replace( /(\d{2})\/(\d{2})\/(\d{4})/, '$3-$2-$1' ); return { ...record, emailValid, dateISO: isoDate, phoneFormatted: `+39 ${record.phone}` }; } exportJSON(records) { return JSON.stringify(records, null, 2); } }
