// Implementazione EventBus class EventBus { constructor() { this.subscribers = {}; } subscribe(event, callback, name) { if (!this.subscribers[event]) { this.subscribers[event] = []; } this.subscribers[event].push({ callback, name }); // Restituisce funzione unsubscribe return () => { this.subscribers[event] = this.subscribers[event] .filter(sub => sub.callback !== callback); }; } publish(event, data) { if (!this.subscribers[event]) return; this.subscribers[event].forEach(sub => { sub.callback(data); }); } } // Utilizzo const eventBus = new EventBus(); // Subscribe eventBus.subscribe('userLoggedIn', (user) => { console.log('User logged in:', user); updateUI(user); }); // Publish eventBus.publish('userLoggedIn', { name: 'Mario' });

// Componenti che usano EventBus class AuthComponent { constructor(eventBus) { this.eventBus = eventBus; this.isLoggedIn = false; } login(email, password) { // Simula autenticazione const user = { email, name: 'Mario Rossi', id: 1 }; this.isLoggedIn = true; // Pubblica evento this.eventBus.publish('user:login', user); } logout() { this.isLoggedIn = false; this.eventBus.publish('user:logout', {}); } } class ProfileComponent { constructor(eventBus) { this.eventBus = eventBus; // Subscribe agli eventi this.eventBus.subscribe('user:login', (user) => { this.loadProfile(user); }); this.eventBus.subscribe('user:logout', () => { this.clearProfile(); }); } loadProfile(user) { console.log('Loading profile:', user); } } // Inizializzazione const bus = new EventBus(); const auth = new AuthComponent(bus); const profile = new ProfileComponent(bus);
