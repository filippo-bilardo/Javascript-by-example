1. Implementazioni Throttle e Debounce

Tecniche per controllare la frequenza di esecuzione delle funzioni.

// THROTTLE: Esegue REGOLARMENTE durante l'evento // Utile per: scroll, resize, mouse move function throttle(func, limit) { let inThrottle; return function(...args) { if (!inThrottle) { func.apply(this, args); inThrottle = true; setTimeout(() => { inThrottle = false; }, limit); } }; } // DEBOUNCE: Esegue DOPO che l'evento si ferma // Utile per: search input, window resize finale, form validation function debounce(func, delay) { let timeout; return function(...args) { clearTimeout(timeout); timeout = setTimeout(() => { func.apply(this, args); }, delay); }; } // Esempi di utilizzo: window.addEventListener('scroll', throttle(() => { console.log('Scrolling...'); // Ogni 200ms }, 200)); searchInput.addEventListener('input', debounce((e) => { searchAPI(e.target.value); // Dopo 500ms di silenzio }, 500));
